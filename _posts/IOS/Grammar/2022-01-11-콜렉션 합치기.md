## 콜렉션 합치기  

~~~
let myFriends = ["철수", "영희", "수잔"]
let otherFriends = ["제임스", "존슨", "존시나"]

// + 더하기를 통해 콜렉션을 합칠 수 있다.
let totalFriends =  myFriends + otherFriends

print(totalFriends)
~~~  

출력
~~~
["철수", "영희", "수잔", "제임스", "존슨", "존시나"]
~~~
