## while 반복문  

~~~
var count = 0
let max = 5
// 5보다 작을 때까지 반복한다.
// while을 통해 특정 조건일 경우에 반복된다.
while count < max {
  count = count + 1
  print("count: ", count)
}
print("5까지 셌다")
~~~  

출력  
~~~
count:  1
count:  2
count:  3
count:  4
count:  5
5까지 셌다
~~~
